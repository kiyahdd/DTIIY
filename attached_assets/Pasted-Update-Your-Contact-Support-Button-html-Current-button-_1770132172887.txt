function openProChatOrRedirect() {
    const isPro = localStorage.getItem('fff_pro_user') === 'true' || 
                  new URLSearchParams(window.location.search).has('test_pro');
    
    if (isPro) {
        // 1. Open Pro Chat Bot
        openProChat();
        
        // 2. Show "Email Human" option in chat
        setTimeout(() => {
            addHumanSupportOption();
        }, 1000);
        
        return;
    }
    
    // Free users go directly to email
    window.location.href = 'mailto:support@falseflagfixer.com?subject=False%20Flag%20Fixer%20Support';
}

function openProChat() {
    const chatWidget = document.getElementById('proChatWidget');
    const chatToggle = document.getElementById('chatToggle');
    
    if (chatWidget) {
        chatWidget.style.display = 'block';
        if (chatToggle) chatToggle.style.display = 'none';
        
        // Auto-add greeting
        setTimeout(() => {
            const messagesDiv = document.getElementById('chatMessages');

function emailHumanSupport() {
    // Get chat history for context
    const chatHistory = getChatHistory();
    const userEmail = localStorage.getItem('user_email') || '';
    
    const subject = 'False Flag Fixer Pro Support Request';
    const body = `Please help with:\n\n` +
                 `Chat History:\n${chatHistory}\n\n` +
                 `User Email: ${userEmail}\n` +
                 `Pro User: Yes\n` +
                 `Timestamp: ${new Date().toISOString()}`;
    
    window.location.href = `mailto:support@falseflagfixer.com?subject=${encodeURIComponent(subject)}&body=${encodeURIComponent(body)}`;
}

// Track support interactions
function trackSupportInteraction(type, resolvedByAI) {
    // Send to analytics
    console.log(`Support ${type}: ${resolvedByAI ? 'AI-resolved' : 'Human-escalated'}`);
    
    // Store for improvement
    const stats = JSON.parse(localStorage.getItem('support_stats') || '{"ai": 0, "human": 0}');
    if (resolvedByAI) stats.ai += 1;
    else stats.human += 1;
    localStorage.setItem('support_stats', JSON.stringify(stats));
}
